CREATE TABLE sqlite_sequence(name,seq);
CREATE TABLE CLASSINFO (ID INTEGER PRIMARY KEY AUTOINCREMENT, CLASSNAME TEXT NOT NULL, PARENT TEXT, INTERFACES TEXT, CHILDREN TEXT);
CREATE TABLE METHODINFO (C_ID INTEGER NOT NULL, METHOD_DESC TEXT NOT NULL, LEVEL INTEGER NOT NULL, FINAL BOOLEAN NOT NULL,LEAF BOOLEAN NOT NULL, WRITTEN_PARAMS TEXT, FOREIGN KEY (C_ID) REFERENCES CLASSINFO(ID));
